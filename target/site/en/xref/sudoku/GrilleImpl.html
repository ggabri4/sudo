127     /**
128      * Vérifie si l'ajout d'une valeur à une position spécifique est possible.
129      *
130      * @param x     Coordonnée x de l'élément.
131      * @param y     Coordonnée y de l'élément.
132      * @param value La valeur à ajouter.
133      * @throws HorsBornesException      Si les coordonnées sont en dehors des
134      *                                  limites de la grille.
135      * @throws ElementInterditException Si l'élément est interdit.
136      * @return True si l'ajout de la valeur est possible, sinon False.
137      */
138     @Override
139     public final boolean isPossible(final int x, final int y, final ElementDeGrille value)
140             throws HorsBornesException, ElementInterditException {
141 
142         if (x < 0 || y < 0 || x >= dimension || y >= dimension) {
143             throw new HorsBornesException("Position en dehors de la grille");
144         }
145 
146         // Vérifier si la valeur existe déjà dans la même ligne
147         for (int i = 0; i < dimension; i++) {
148             if (grille[x][i] != null && grille[x][i].equals(value)) {
149                 return false;
150             }
151         }
152 
153         // Vérifier si la valeur existe déjà dans la même colonne
154         for (int i = 0; i < dimension; i++) {
155             if (grille[i][y] != null && grille[i][y].equals(value)) {
156                 return false;
157             }
158         }
159 
160         // Vérifier si la valeur existe déjà dans le même carré
161         int squareSize = (int) Math.sqrt(dimension);
162         int startX = (x / squareSize) * squareSize;
163         int startY = (y / squareSize) * squareSize;
164 
165         for (int i = startX; i < startX + squareSize; i++) {
166             for (int j = startY; j < startY + squareSize; j++) {
167                 if (grille[i][j] != null && grille[i][j].equals(value)) {
168                     return false;
169                 }
170             }
171         }
172         return true;
173     }
